Database: company_db
Tables:

Employee

emp_id INT PRIMARY KEY AUTO_INCREMENT,
name VARCHAR(50),
email VARCHAR(100),
salary DOUBLE,
dept_id INT,
joining_date DATE


Department

dept_id INT PRIMARY KEY AUTO_INCREMENT,
dept_name VARCHAR(50),
location VARCHAR(50)
